Write a Python function that takes a number n and returns a list of the first n Fibonacci numbers. Plot the Fibonacci sequence using a line graph.

import matplotlib.pyplot as plt
def fibonacci(n):
    if n <= 0:
        return "Incorrect Input"
    series = [0, 1]
    if n > 2:
        for i in range(2, n):
            series.append(series[i-1] + series[i-2])
    return series

n = int(input("Enter num"))
series = fibonacci(n) #this returns array of fibonacci numbers
plt.plot(list(range(1,n+1)),series)
plt.xlabel("Input")  
plt.ylabel("Fibonacci Series")  
plt.title("Fibonacci Graph")
plt.show()

import random
import matplotlib.pyplot as plt
randomNums = []
for i in range(100):
    randomNums.append(random.randint(1,50))
file = open('data2.txt','w+')
for randomNum in randomNums:
    file.write(f'{randomNum}\n')
file.close()
file = open('data2.txt','r')
content = file.read()
numS = content.split('\n')
sumNums = 0
dictNumFreq = {}
for num in numS:
    if len(num) != 0:
        sumNums = sumNums + int(num)
        if num in dictNumFreq:
            dictNumFreq[num] += 1
        else:
            dictNumFreq[num] = 1
